树的遍历

给出一棵树的中序和后序遍历，求层序遍历。

``` C++
#include <iostream>
#include <queue>
#include <algorithm>
#include <cstdio>
using namespace std;
// 范神  Orz
int val[1005],lid[1005],rid[1005],a[1005],b[1005],cnt;
int n;
int fuck(long l,long r,long lb,long rb)     //前两个参数为在中序遍历中找下一个父节点的区间，后两个参数为后续遍历数组的区间，后者同时为下一个父节点在后序遍历中的位置
{
    if (l>r || lb>rb)                       //验证是否存在有效区间
        return 0;
    int t = cnt++;                          //有效遍历的次数即为节点的个数，前序遍历的序数
    val[t] = b[rb];                         //将后序遍历中最后的父节点即最后一个点存入父节点的集合，val[]存储的是树的前序遍历
    long ans = find(a+l,a+r+1,val[t]) - a;  //找到父节点在中序遍历中的位置
    if (ans == r+1)                         //找不到立即返回
        return t;
    lid[t]=fuck(l,ans-1,lb,lb+ans-l-1);     //在中序遍历的左半边继续搜索，依次记录前序遍历中左节点的序数
    rid[t]=fuck(ans+1,r,lb+ans-l,rb-1);     //在中序遍历的右半边继续搜索，依次记录前序遍历中右节点的序数
    return t;
}
int main()
{
    cin>>n;
    for (int i=0; i<n; i++)         //后续遍历
        cin>>b[i];
    for (int i=0; i<n; i++)         //中序遍历
        cin>>a[i];
    fuck(0,n-1,0,n-1);              //前序遍历
    int num=0;
    queue<int> q;
    q.push(0);
    while(!q.empty())
    {
        num++;
        int x=q.front();
        q.pop();
        printf("%d%c",val[x]," \n"[num==n]);    //一个卖萌的小把戏，将空格和换行存在一个字符数组中，根据条件输出数组的首位或第二位
        if (lid[x]) q.push(lid[x]);             //将左节点的下一个左节点（若存在）推进队列，注意在此之前同层右节点已被推入队列
        if (rid[x]) q.push(rid[x]);
    }
    return 0;
}
/* 
 样例：
 7
 2 3 1 5 7 6 4
 1 2 3 4 5 6 7
 给出样例的的遍历过程中fuck()四个参数的变化：
 0 6 0 6
 0 2 0 2
 0 -1 0 -1
 1 2 0 1
 1 1 0 0
 1 0 0 -1
 2 1 0 -1
 3 2 1 0
 4 6 3 5
 4 4 3 3
 4 3 3 2
 5 4 3 2
 6 6 4 4
 6 5 4 3
 7 6 4 3
 */

大概思路是按前序遍历树，模拟链表纪录下节点的顺序。
